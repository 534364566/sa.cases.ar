    <script>
        const apiKey = "";
        const ASESORES = { 'Santino': '5491150509892', 'Agustina': '5491123188640' };
        let cart = { magsafe: 0, silicone: 0, vidrio: 0 };

        function updateQty(item, change) {
            cart[item] = Math.max(0, cart[item] + change);
            document.getElementById(`qty-${item}`).innerText = cart[item];
            calcularTotal();
        }

        function calcularTotal() {
            let total = 0;
            let promoMsg = "";
            let items = {...cart}; 
            while(items.magsafe >= 1 && items.vidrio >= 1) { total += 13900; items.magsafe--; items.vidrio--; promoMsg = "¡Combo aplicado! ✨"; }
            while(items.silicone >= 3) { total += 25500; items.silicone -= 3; promoMsg = "¡Combo 3 Silicones aplicado! ✨"; }
            total += items.magsafe * 9900; total += items.silicone * 8900; total += items.vidrio * 4500;
            const section = document.getElementById('promoSection');
            if(total > 0) {
                section.classList.remove('hidden');
                document.getElementById('totalPrice').innerText = `$${total.toLocaleString()}`;
                document.getElementById('promoText').innerText = promoMsg;
            } else { section.classList.add('hidden'); }
        }

        function mostrarModal() { document.getElementById('contactModal').classList.remove('hidden'); }
        function cerrarModal() { document.getElementById('contactModal').classList.add('hidden'); }

        function abrirWhatsApp(nombreAsesor) {
            const numero = ASESORES[nombreAsesor];
            const modelo = document.getElementById('modelSelect').value;
            let itemsElegidos = [];
            if (cart.magsafe > 0) itemsElegidos.push(`${cart.magsafe} MagSafe`);
            if (cart.silicone > 0) itemsElegidos.push(`${cart.silicone} Silicone`);
            if (cart.vidrio > 0) itemsElegidos.push(`${cart.vidrio} Vidrio(s)`);

            let mensaje = `Hola ${nombreAsesor}! ✨ Vengo de sa.cases.ar. Busco asesoramiento emocional para mi iPhone.`;
            if (modelo) mensaje += `\n📱 Modelo: ${modelo}.`;
            if (itemsElegidos.length > 0) {
                mensaje += `\n🛒 Elegí: ${itemsElegidos.join(', ')}.`;
                mensaje += `\n💰 Total estimado: ${document.getElementById('totalPrice').innerText}.`;
            } else { mensaje += `\n🤔 Quiero que me ayuden a elegir mi funda ideal.`; }
            window.open(`https://wa.me/${numero}?text=${encodeURIComponent(mensaje)}`, '_blank');
            cerrarModal();
        }

        async function askGemini() {
            const input = document.getElementById('aiInput');
            const resBox = document.getElementById('aiResponse');
            const btn = document.getElementById('aiBtn');
            const texto = input.value.trim();
            if(!texto) return;
            btn.disabled = true; resBox.classList.remove('hidden'); resBox.innerHTML = "Pensando... ✨";
            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ contents: [{ parts: [{ text: `Asesor experto de sa.cases.ar. Sugiere funda para: "${texto}". Menciona a Santino y Agustina y el valor emocional.` }] }] })
                });
                const data = await response.json();
                resBox.innerHTML = `"${data.candidates[0].content.parts[0].text}"`;
            } catch (error) { resBox.innerHTML = "¡Escribinos por WhatsApp!"; } 
            finally { btn.disabled = false; }
        }
    </script>
